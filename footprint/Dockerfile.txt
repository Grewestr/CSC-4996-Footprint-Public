# Dockerfile
FROM python:3.10-slim

WORKDIR /app

# Install dependencies
RUN apt-get update && apt-get install -y --no-install-recommends \
    libgl1-mesa-glx \
    libglib2.0-0 \
    && pip install --no-cache-dir torch opencv-python-headless numpy yt-dlp redis rq inference_sdk pandas \
    && apt-get clean && rm -rf /var/lib/apt/lists/*

# Copy app code
COPY . .

# Environment variable for Redis
ENV REDIS_URL=redis://redis-server:6379/0

# Start worker with scheduler
CMD ["rq", "worker", "--with-scheduler"]
